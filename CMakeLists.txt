cmake_minimum_required(VERSION 3.21)
project(CANtrip CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

if(NOT WIN32)
    message(FATAL_ERROR "CANtrip targets Windows only (its CAN backends and the Wireshark extcap plumbing are Windows-specific).")
endif()

list(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_SOURCE_DIR}/cmake")
include(CANtripCompilerOptions)

option(CANTRIP_BUILD_APP "Build the CANtrip Qt desktop app (requires Qt6)" ON)

# dbcppp's NS_: (new symbols) grammar (third_party/dbcppp/src/DBCX3.cpp) only
# recognizes a fixed, case-sensitive whitelist of keyword strings - it accepts
# "FILTER" but not "Filter". Real-world DBC files generated by PSA's DBC
# tooling (confirmed against actual files) emit "Filter" (mixed case) in the
# NS_: symbol list. Since that list is matched by a non-backtracking Kleene
# star, one unrecognized symbol silently truncates the whole NS_: section
# instead of failing right there - the parser then reports a confusing
# "Expecting: BitTiming" error several tokens later, and the DBC is rejected
# outright even though it's otherwise well-formed. Patched here (not by
# editing the submodule's checked-out file directly) so the fix survives a
# fresh `git submodule update` - idempotent, so it only touches the file if
# the mixed-case variant isn't already present.
set(_dbcx3_file "${CMAKE_CURRENT_SOURCE_DIR}/third_party/dbcppp/src/DBCX3.cpp")
file(READ "${_dbcx3_file}" _dbcx3_contents)
if(NOT _dbcx3_contents MATCHES "string\\(\"Filter\"\\)")
    string(REPLACE
        "string(\"FILTER\")[SetVal]"
        "string(\"FILTER\")[SetVal]\n            | string(\"Filter\")[SetVal]"
        _dbcx3_contents "${_dbcx3_contents}")
    file(WRITE "${_dbcx3_file}" "${_dbcx3_contents}")
endif()

# dbcppp only needs to parse .dbc files for us - skip its KCD (XML) support,
# CLI tool, and test/example targets so we don't drag in libxml2/cxxopts.
set(build_kcd OFF CACHE BOOL "" FORCE)
set(build_tools OFF CACHE BOOL "" FORCE)
set(build_tests OFF CACHE BOOL "" FORCE)
set(build_examples OFF CACHE BOOL "" FORCE)
add_subdirectory(third_party/dbcppp)
# dbcppp's own src/CMakeLists.txt (which sets this) isn't actually used by
# its root CMakeLists.txt - a quirk in the upstream project, not us being
# clever - so without this every exported symbol stays __declspec(dllimport)
# even while building the DLL itself, which MSVC rightly refuses to link.
target_compile_definitions(dbcppp PRIVATE DBCPPP_EXPORT)

add_subdirectory(common)
add_subdirectory(extcap)

if(CANTRIP_BUILD_APP)
    add_subdirectory(app)
endif()
